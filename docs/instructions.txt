Note: words inside paranthesis "()" are optional

[x] nop;    0x00;   no operation

[x] stc;    ----;   
[x] sti;    0x01;   store (u)int constant on stack
[x] stf;    0x02;   store float constant on stack
[x] stb;    0x03;   store (u)byte constant on stack

[x] ldc;    ----;   
[x] ldi;    0x04;   load (u)int from stack on heap and set &eax to address 
[x] ldf;    0x05;   load float from stack on heap and set &eax to address 
[x] ldb;    0x06;   load (u)byte from stack on heap and set &eax to address 

[x] rda;    ----;   
[x] rdi;    0x07;   read (u)int from heap on address &<given_stack> and store it on stack
[x] rdf;    0x08;   read float from heap on address &<given_stack> and store it on stack
[x] rdb;    0x09;   read (u)byte from heap on address &<given_stack> and store it on stack
[x] rdr;    0x0A;   read the &<given_register>

[x] mov;    ----;   
[x] movi;   0x0B;   set register to given int
[x] movf;   0x0C;   set register to given float 
[x] movb;   0x0D;   set register to given byte 
[x] movr;   0x0E;   set register to value at given register

[ ] add;    ----;
